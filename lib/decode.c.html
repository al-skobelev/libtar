30char *
31th_get_pathname(TAR *t)
32{
33        char filename[MAXPATHLEN];
34
35        if (t->th_buf.gnu_longname)
36                return strdup(t->th_buf.gnu_longname);
37
38        if (t->th_buf.prefix[0] != '\0')
39        {
40                snprintf(filename, sizeof(filename), "%.155s/%.100s",
41                         t->th_buf.prefix, t->th_buf.name);
42                return strdup(filename);
43        }
44

50uid_t
51th_get_uid(TAR *t)
52{
53        int uid;
54        struct passwd *pw;
55
56        pw = getpwnam(t->th_buf.uname);
57        if (pw != NULL)
58                return pw->pw_uid;
59
60        /* if the password entry doesn't exist */
61        sscanf(t->th_buf.uid, "%o", &uid);
62        return uid;
63}

82mode_t
83th_get_mode(TAR *t)
84{
85        mode_t mode;
86
87        mode = (mode_t)oct_to_int(t->th_buf.mode);
88        if (! (mode & S_IFMT))
89        {
90                switch (t->th_buf.typeflag)
91                {
92                case SYMTYPE:
93                        mode |= S_IFLNK;
94                        break;
95                case CHRTYPE:
96                        mode |= S_IFCHR;
97                        break;
98                case BLKTYPE:
99                        mode |= S_IFBLK;
100                        break;
101                case DIRTYPE:
102                        mode |= S_IFDIR;
103                        break;
104                case FIFOTYPE:
105                        mode |= S_IFIFO;
106                        break;
107                case AREGTYPE:
108                        if (t->th_buf.name[strlen(t->th_buf.name) - 1] == '/')
109                        {
110                                mode |= S_IFDIR;
111                                break;
112                        }
113                        /* FALLTHROUGH */
114                case LNKTYPE:
115                case REGTYPE:
116                default:
117                        mode |= S_IFREG;
118                }
119        }
120
121        return mode;
122}
